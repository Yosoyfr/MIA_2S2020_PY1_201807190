//Test de un disco
func TestDisc(path string) {
	//Obtenemos el mbr del disco
	file, mbr := readFile(path)
	defer file.Close()
	fmt.Println(mbr)

	//Llenamos una particion
	var aux [300]byte

	file.Seek(2175, 0)
	var binaryDisc bytes.Buffer
	binary.Write(&binaryDisc, binary.BigEndian, &aux)
	writeNextBytes(file, binaryDisc.Bytes())

}
